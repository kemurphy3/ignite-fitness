LoadCalculator.getCurrentLoadStatus() - Load Analysis

getCurrentLoadStatus() { // Calculate current week's load const weeklyLoad = this.calculateWeeklyLoad(sessions); const externalLoad = this.calculateExternalLoad(activities); const totalCurrentLoad = weeklyLoad.totalLoad + externalLoad.totalLoad; // Calculate 7-day average for comparison const sevenDayAverage = this.calculateSevenDayAverage(sessions, activities); // Detect load spikes const loadSpike = this.detectLoadSpike(totalCurrentLoad, sevenDayAverage); // Generate workout intensity recommendations const workoutRecommendations = this.generateWorkoutIntensityRecommendations( totalCurrentLoad, sevenDayAverage, loadSpike ); return { currentLoad: totalCurrentLoad, sevenDayAverage: sevenDayAverage, loadSpike: loadSpike, recommendations: workoutRecommendations }; }

LoadCalculator.generateWorkoutIntensityRecommendations() - Intensity Scaling

generateWorkoutIntensityRecommendations(currentLoad, sevenDayAverage, loadSpike) { const loadRatio = sevenDayAverage > 0 ? currentLoad / sevenDayAverage : 1.0; let intensity = 1.0; let volume = 1.0; let message = 'Normal training load'; // High weekly load (>7 day average) reduces volume by 20% if (loadRatio > 1.0) { volume = 0.8; // 20% reduction message = 'Reduced volume due to high training load'; if (loadRatio > 1.3) { volume = 0.6; // 40% reduction intensity = 0.8; // 20% intensity reduction message = 'Significantly reduced volume and intensity due to high training load'; } } // Load spike detection triggers recovery day recommendation if (loadSpike.isSpike && loadSpike.severity === 'high') { recoveryRecommended = true; intensity = 0.5; volume = 0.3; message = 'Recovery day recommended due to load spike'; } return { intensity, volume, message, recoveryRecommended }; }

ExpertCoordinator.applyLoadBasedWorkoutAdjustments() - Context Integration

applyLoadBasedWorkoutAdjustments(context, loadStatus) { const recommendations = loadStatus.recommendations; // Apply intensity and volume adjustments if (recommendations.intensity < 1.0) { context.intensityScale *= recommendations.intensity; context.loadAdjustments.push(`Load-based intensity reduction: ${recommendations.adjustments.intensityReduction}%`); } if (recommendations.volume < 1.0) { context.volumeScale = recommendations.volume; context.loadAdjustments.push(`Load-based volume reduction: ${recommendations.adjustments.volumeReduction}%`); } // Add load rationale to context context.loadAdjustments.push(recommendations.message); // Set recovery recommendation flag if (recommendations.recoveryRecommended) { context.recommendRecoveryDay = true; } }

  • 7-Day Average: Historical average load for comparison
  • Load Ratio: currentLoad / sevenDayAverage
  • Spike Detection: Ratio > 1.1 = spike, > 1.5 = high spike

adjustRepsForVolume(reps, volumeMultiplier) { if (typeof reps === 'string') { // Handle range strings like "8-12" const match = reps.match(/(\d+)-(\d+)/); if (match) { const min = Math.max(1, Math.floor(parseInt(match[1]) * volumeMultiplier)); const max = Math.max(1, Math.floor(parseInt(match[2]) * volumeMultiplier)); return `${min}-${max}`; } } if (typeof reps === 'number') { return Math.max(1, Math.floor(reps * volumeMultiplier)); } return reps; }
